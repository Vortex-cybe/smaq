export async function onRequestPost({ request, env }) { try { const data = await request.json(); // validation minimale if (!data?.name || !data?.email || !data?.need || !data?.msg) { return new Response(JSON.stringify({ ok:false, error:"missing_fields" }), { status: 400, headers: { "Content-Type":"application/json" } }); } const id = crypto.randomUUID(); const key = `lead:${Date.now()}:${id}`; // KV optionnel if (env.LEADS_KV) { await env.LEADS_KV.put(key, JSON.stringify(data), { expirationTtl: 60 * 60 * 24 * 365 }); } return new Response(JSON.stringify({ ok:true, id }), { status: 200, headers: { "Content-Type":"application/json" } }); } catch (e) { return new Response(JSON.stringify({ ok:false, error:"bad_request" }), { status: 400, headers: { "Content-Type":"application/json" } }); } }
